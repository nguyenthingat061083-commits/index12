        app.style.setProperty('--accent', color);

            n.style.borderLeftColor = color;
        });
        // Cập nhật switch checked
        document.querySelectorAll('.switch input:checked + .slider').forEach(s => {
            s.style.background = color;
        });
        // Cập nhật icon
        document.querySelectorAll('.logo i, .nav-item i, .card h3 i, .row .label i, .section-title i').forEach(el => {
            el.style.color = color;

            s.style.outlineColor = color;
        });
        // Cập nhật slider thumb
        document.querySelectorAll('input[type="range"]::-webkit-slider-thumb').forEach(t => {
            t.style.background = color;
        });
    }

        document.querySelectorAll('.nav-label').forEach((el, i) => {
            if (i < dict['nav-label'][lang].length) {
                el.textContent = dict['nav-label'][lang][i];
            }
        });

        // Cập nhật section title (h3)
        document.querySelectorAll('.sec-title').forEach((el, i) => {
            if (i < dict['sec-title'][lang].length) {
                el.textContent = dict['sec-title'][lang][i];
            }
        });

        // Cập nhật description (p .desc)
        document.querySelectorAll('.desc').forEach((el, i) => {
            if (i < dict['desc'][lang].length) {
                el.textContent = dict['desc'][lang][i];
            }
        });

        // Cập nhật label (.lbl)
        const labels = document.querySelectorAll('.lbl');
        let idx = 0;
        labels.forEach(el => {
            if (idx < dict['lbl'][lang].length) {
                el.textContent = dict['lbl'][lang][idx];
                idx++;
            }
        });
    }

    // ===== KHỞI TẠO THEME MẶC ĐỊNH =====
    document.getElementById('appContainer').style.setProperty('--accent', '#ff6b2c');

    // ===== KHỞI TẠO NGÔN NGỮ =====
    // Mặc định English
    document.getElementById('langSelect').value = 'en';
    switchLanguage();
</script>
